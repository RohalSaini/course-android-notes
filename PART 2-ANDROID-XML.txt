class MainActivity : AppCompatActivity() {
    lateinit var editText_username: EditText    // telling compiler to have varible of TYPE: Edittext
    lateinit var editText_password: EditText   // telling compiler to have varible of TYPE: Edittext
    lateinit var editText_emailAddress: EditText   // telling compiler to have varible of TYPE: Edittext
    lateinit var editText_city:EditText   // telling compiler to have varible of TYPE: Edittext

    lateinit var button_Login:Button   // telling compiler to have varible of TYPE: Button
    // lateinit means it will intlialze late



    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
       
        editText_username = findViewById(R.id.editext_username)   // binding ID with varible
        editText_emailAddress = findViewById(R.id.editext_emailAddress)  // binding ID with varible
        editText_password = findViewById(R.id.editext_passsword)  // binding ID with varible
        editText_city = findViewById(R.id.editext_city)  // binding ID with varible

        button_Login = findViewById(R.id.button_login) // binding ID with varible


        button_Login.setOnClickListener {
        
            var username = editText_username.text   // Getting value from Editext
            var password = editText_password.text  // Getting value from Editext
            var email = editText_emailAddress.text  // Getting value from Editext
            var city = editText_city.text   // Getting value from Editext

            println(" Name : $username  \n Password: $password  \n Email : $email  \n City: $city")

            Toast.makeText(this," Name : $username  \n Password: $password  \n Email : $email  \n City: $city",Toast.LENGTH_LONG).show()
        }
    }
}

class MainActivity : AppCompatActivity() {
    lateinit var editText_username: EditText
    lateinit var editText_password: EditText
    lateinit var editText_emailAddress: EditText
    lateinit var editText_city:EditText
    lateinit var button_Login:Button

    lateinit var checbox_updates: CheckBox

    lateinit var radioButton_gender: RadioButton
    lateinit var radioGroup :RadioGroup

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
        editText_username = findViewById(R.id.editext_username)
        editText_emailAddress = findViewById(R.id.editext_emailAddress)
        editText_password = findViewById(R.id.editext_passsword)
        editText_city = findViewById(R.id.editext_city)
        button_Login = findViewById(R.id.button_login)
        checbox_updates = findViewById(R.id.checkbox_updates)

        radioGroup=findViewById(R.id.radioGroup)

        button_Login.setOnClickListener {

            var username = editText_username.text
            var password = editText_password.text
            var email = editText_emailAddress.text
            var city = editText_city.text
            var gender= ""
            var updates = "No"
            if(checbox_updates.isChecked) {
                updates = "Yes"
            }
            val selectedId = radioGroup.checkedRadioButtonId
            radioButton_gender = findViewById(selectedId)
            if (selectedId == -1) {
                Toast.makeText(this, "Please select Gender", Toast.LENGTH_SHORT).show()
            } else {
                    gender = radioButton_gender.text.toString()
            }
            println(" Name : $username  \n Password: $password  \n Email : $email  \n City: $city \n want updates: $updates \n Gender: $gender")
            Toast.makeText(this," Name : $username  \n Password: $password  \n Email : $email  \n City: $city \n want updates: $updates \n Gender: $gender",Toast.LENGTH_LONG).show()
        }
    }
}

        val aa:ArrayAdapter<String> = ArrayAdapter(this,android.R.layout.simple_spinner_item,country)

        this.city = country[0]


        button_Login.setOnClickListener {

            var username = editText_username.text
            var password = editText_password.text
            var email = editText_emailAddress.text
            var gender= ""

class MainActivity : AppCompatActivity(), AdapterView.OnItemSelectedListener {
    lateinit var editText_username: EditText
    lateinit var editText_password: EditText
    lateinit var editText_emailAddress: EditText
    lateinit var button_Login:Button

    lateinit var checbox_updates: CheckBox

    lateinit var radioButton_gender: RadioButton
    lateinit var radioGroup :RadioGroup

    lateinit var spinner_city:Spinner
    lateinit var countries:Array<String>
    lateinit var city:String


    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
        editText_username = findViewById(R.id.editext_username)
        editText_emailAddress = findViewById(R.id.editext_emailAddress)
        editText_password = findViewById(R.id.editext_passsword)
        button_Login = findViewById(R.id.button_login)
        checbox_updates = findViewById(R.id.checkbox_updates)
        radioGroup=findViewById(R.id.radioGroup)

        countries = resources.getStringArray(R.array.cities_array)

        spinner_city = findViewById(R.id.editext_city)
        spinner_city.onItemSelectedListener= this

        ArrayAdapter(this,android.R.layout.simple_spinner_item,countries)
            .also {
                    adapter ->  spinner_city.adapter = adapter
             }

        button_Login.setOnClickListener {

            var username = editText_username.text
            var password = editText_password.text
            var email = editText_emailAddress.text
            var gender= ""
            var updates = "No"
            if(checbox_updates.isChecked) {
                updates = "Yes"
            }
            val selectedId = radioGroup.checkedRadioButtonId
            radioButton_gender = findViewById(selectedId)
            if (selectedId == -1) {
                Toast.makeText(this, "Please select Gender", Toast.LENGTH_SHORT).show()
            } else {
                    gender = radioButton_gender.text.toString()
            }
            println(" Name : $username  \n Password: $password  \n Email : $email  \n City: $city \n want updates: $updates \n Gender: $gender")
            Toast.makeText(this," Name : $username  \n Password: $password  \n Email : $email  \n City: $city \n want updates: $updates \n Gender: $gender",Toast.LENGTH_LONG).show()
        }
    }

    override fun onItemSelected(parent: AdapterView<*>?, p1: View?, position: Int, p3: Long) {
       ///  Toast.makeText(this,countries[position] , Toast.LENGTH_LONG).show();
          this.city = countries[position]
    }

        TODO("Not yet implemented")
    }
